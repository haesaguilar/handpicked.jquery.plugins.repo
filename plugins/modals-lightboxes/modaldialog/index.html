Error

Warning

Success

Prompt

Set Title

  });

Dialog Width 
$.modaldialog.success('Not much here.', {
  width: 200
  });

AutoClose 
$.modaldialog.success('Autoclosing in 5 seconds.', {
  timeout: 5
  });

AutoClose

  timeout: 5

  });
